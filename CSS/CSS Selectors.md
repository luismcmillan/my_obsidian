[[CSS Cascading Style Sheets]]

```CSS
h1 {    #Element Selector
	color: blue
}





.class-1234 {    #Class Selector
	color: blue
}
<h2 class="class-1234"> Content </h2>





#main {    #Id Selector
	color: blue
}
<h2 id="main"> Content </h2>



p[draggable]{    #Attribute Selector
	color: blue
}
<h2 draggable="true"> Content </h2>

p[draggable="false"]{    #Attribute Selector
	color: blue
}
<h2 draggable="false"> Content </h2>


* {    #Universal Selector
	color: blue
}

```
