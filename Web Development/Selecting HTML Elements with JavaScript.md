```JavaScript
document.getElementsByTagName(...) //return array

document.getElementsByTagName(...).length 	//returns Array.size					  
document.getElementsByClassName(...) //returns array

document.getElementById(...) //return Item

document.querySelector("h1") //return h1 item

document.querySelector("#title") //return item with ID title

document.querySelector(".btn") //return item with class=btn

document.querySelector("li a") //return anchor which is List-item li

document.querySelector("li.lala") //return list item which is of class=lala

document.querySelectorAll("#list .item") //return Array of elements of list with ID=list and class=item
```
